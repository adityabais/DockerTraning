apiVersion: v1
kind: ReplicationController
metadata:
  name: nginx
spec:
  replicas: 3
  selector:
    app: nginx
  template:
    metadata:
      name: nginx
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx
        ports:
        - containerPort: 80


Command to run Replication Controller:
kubectl apply -f replication.yaml
kubectl get pods -o wide
kubectl describe replicationcontrollers/nginx
